SELECT * FROM SQLEAD569.VIEW1

CREATE VIEW SQLEAD569.VIEW2 AS SELECT IDFUNC,NOMEFUNC FROM SQLEAD569.FUNC

CREATE VIEW SQLEAD569.VIEW3 AS SELECT T1.IDFUNC,T1.NOMEFUNC,T2.PROJETO FROM
            SQLEAD569.FUNC T1 INNER JOIN SQLEAD569.PROJETO T2 ON
            T1.IDFUNC = T2.IDFUNC
